from fastapi import FastAPI, Request
from fastapi.responses import HTMLResponse
from fastapi.staticfiles import StaticFiles
from fastapi.templating import Jinja2Templates
from pydantic import BaseModel

import sqlite3
from datetime import datetime, timezone

DB_PATH = "app.db"

app = FastAPI(title="RPi Motion Server")

# serve /static/*
app.mount("/static", StaticFiles(directory="static"), name="static")
templates = Jinja2Templates(directory="templates")


def init_db():
    with sqlite3.connect(DB_PATH) as con:
        con.execute("""
        CREATE TABLE IF NOT EXISTS visits (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            ts TEXT NOT NULL,
            ip TEXT,
            user_agent TEXT
        );
        """)
        con.execute("""
        CREATE TABLE IF NOT EXISTS motion_events (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            ts TEXT NOT NULL,
            device TEXT,
            event TEXT,
            motion INTEGER,
            rssi INTEGER,
            millis INTEGER,
            source_ip TEXT
        );
        """)
        con.commit()


@app.on_event("startup")
def on_startup():
    init_db()


def log_visit(ip: str, user_agent: str):
    ts = datetime.now(timezone.utc).isoformat()
    with sqlite3.connect(DB_PATH) as con:
        con.execute(
            "INSERT INTO visits(ts, ip, user_agent) VALUES (?, ?, ?)",
            (ts, ip, user_agent),
        )
        con.commit()


def log_motion_event(device: str, event: str, motion: bool, rssi: int | None, millis: int | None, source_ip: str):
    ts = datetime.now(timezone.utc).isoformat()
    with sqlite3.connect(DB_PATH) as con:
        con.execute(
            "INSERT INTO motion_events(ts, device, event, motion, rssi, millis, source_ip) VALUES (?, ?, ?, ?, ?, ?, ?)",
            (ts, device, event, 1 if motion else 0, rssi, millis, source_ip),
        )
        con.commit()


def get_stats(visits_limit: int = 20, motion_limit: int = 20):
    with sqlite3.connect(DB_PATH) as con:
        cur = con.cursor()
        cur.execute("SELECT COUNT(*) FROM visits")
        total_visits = cur.fetchone()[0]

        cur.execute("SELECT COUNT(*) FROM motion_events")
        total_motion = cur.fetchone()[0]

        cur.execute(
            "SELECT ts, ip, user_agent FROM visits ORDER BY id DESC LIMIT ?",
            (visits_limit,),
        )
        visit_rows = cur.fetchall()

        cur.execute(
            "SELECT ts, device, event, motion, rssi, millis, source_ip FROM motion_events ORDER BY id DESC LIMIT ?",
            (motion_limit,),
        )
        motion_rows = cur.fetchall()

    latest_visits = [{"ts": ts, "ip": ip or "-", "ua": (ua or "-")} for (ts, ip, ua) in visit_rows]
    latest_motion = [
        {
            "ts": ts,
            "device": device or "-",
            "event": event or "-",
            "motion": bool(motion),
            "rssi": rssi,
            "millis": millis,
            "source_ip": source_ip or "-",
        }
        for (ts, device, event, motion, rssi, millis, source_ip) in motion_rows
    ]
    return total_visits, total_motion, latest_visits, latest_motion


class MotionEvent(BaseModel):
    device: str = "nano33iot"
    event: str
    motion: bool
    rssi: int | None = None
    millis: int | None = None


@app.get("/", response_class=HTMLResponse)
def home(request: Request):
    # NOTE: if you're behind a reverse proxy, you may want to read X-Forwarded-For instead.
    ip = request.client.host if request.client else "unknown"
    ua = request.headers.get("user-agent", "-")

    log_visit(ip, ua)
    total_visits, total_motion, latest_visits, latest_motion = get_stats()

    return templates.TemplateResponse(
        "index.html",
        {
            "request": request,
            "total_visits": total_visits,
            "total_motion": total_motion,
            "latest_visits": latest_visits,
            "latest_motion": latest_motion,
        },
    )


@app.post("/motion")
def motion_endpoint(request: Request, e: MotionEvent):
    source_ip = request.client.host if request.client else "unknown"
    log_motion_event(e.device, e.event, e.motion, e.rssi, e.millis, source_ip)
    return {"ok": True, "received": e.event}


@app.get("/api/motion/latest")
def motion_latest(limit: int = 20):
    # simple JSON endpoint (useful for dashboards)
    _, _, _, latest_motion = get_stats(motion_limit=limit)
    return {"items": latest_motion}
